ДЛЯ АВТОЗАПУСКА СКРИПТА
1. sudo nano /etc/systemd/system/your_script.service
2.
[Unit]
Description=Python script autostart

[Service]
ExecStartPre=/bin/sleep 5
ExecStart=/usr/bin/python3 /path/to/your_script.py
WorkingDirectory=/path/to/
Restart=always
User=your_username

[Install]
WantedBy=multi-user.target
3. sudo systemctl enable your_script.service
4. sudo systemctl start your_script.service

Для перезапуска:
1. sudo systemctl daemon-reload
2. sudo systemctl restart your_script.service

ДЛЯ ROOT ПОРТОВ
1. sudo nano /etc/systemd/system/chmod_usb.service
2.
[Unit]
Description=Set permissions for /dev/ttyUSB*

[Service]
Type=oneshot
ExecStart=/bin/bash -c 'chmod 777 /dev/ttyUSB0 && chmod 777 /dev/ttyUSB1'

[Install]
WantedBy=multi-user.target
3. sudo systemctl enable chmod_usb.service
4. sudo systemctl start chmod_usb.service

Для проверки: ls -l /dev/ttyUSB*
crwxrwxrwx - это root
